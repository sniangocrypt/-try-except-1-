#Задание: Ошибка с файлами

#Создайте программу, которая:

#	1.	Читает файл по указанному пути.
#	2.	Обрабатывает исключение FileNotFoundError в случае, если файл не найден, и выводит соответствующее сообщение.

#Требования:

#	1.	Программа должна запросить у пользователя путь к файлу.
#
#	2.	Если файл не найден, необходимо вывести сообщение: "Файл по указанному пути не найден. Пожалуйста, проверьте путь и повторите попытку."

#	3.	Используйте конструкцию try...except для обработки исключения.


info = str(input(f"Введитe путь к файлу :"))
try:
    with open(info, "r", encoding="utf-8") as file:
        data = file.readlines()  # Читаем содержимое файла
        print (data)
except FileNotFoundError:
    print(f"Файл по указанному пути не найден. Пожалуйста, проверьте путь и повторите попытку.")
